# 2. ONE LINE INSTALLATION FOR TERMUX

pkg update -y && pkg install python git -y && git clone https://github.com/https-penguin/penguinc.git && cd penguinc && python -m venv venv && source venv/bin/activate && pip install colorama termcolor && python main.py

cd penguinc
source venv/bin/activate
python main.py


# 3. ONE LINE INSTALLATION FOR LINUX

sudo apt update -y && sudo apt install python3 python3-venv git -y && git clone https://github.com/https-penguin/penguinc.git && cd penguinc && python3 -m venv venv && source venv/bin/activate && pip install colorama termcolor && python main.py

cd penguinc && source venv/bin/activate && python main.py
